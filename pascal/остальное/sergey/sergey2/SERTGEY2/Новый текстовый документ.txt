        private void bt1_Click(object sender, EventArgs e)
        {
            int sum = Convert.ToInt16(tb1.Text) + Convert.ToInt16(tb2.Text);
            int mul = Convert.ToInt16(tb1.Text) * Convert.ToInt16(tb2.Text);
            int nod = 1;

            int a = Convert.ToInt16(tb1.Text);
            int b = Convert.ToInt16(tb2.Text);
            int Max = 0;

            if (a > b) {Max = a;} else {Max = b;}

            for (int j = Max; j >= 1; j--)
            {
                if ((a % j == 0) && (b % j == 0)) { nod = j; break; }

            }
            if (cb1.Checked)
            {
                MessageBox.Show("Результат = " + Convert.ToString(sum), "Cумма");
            }
            if (cb2.Checked)
            {
                MessageBox.Show("Результат = " + Convert.ToString(mul),"Умножить");
            }
            if (cb3.Checked)
            {
                MessageBox.Show("Результат = " + Convert.ToString(nod), "Наибольший общий делитель");
            }
           
        }
